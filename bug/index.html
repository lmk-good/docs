# Bug总结

WARNING

Element-UIForm表单中Select下拉框自带问题

1、首先创建一个js文件，操作如下:

let lock = true;
let el = null;
// const MousedownEvent = new Event('mousedown', { bubbles: true });
var MousedownEvent = document.createEvent('HTMLEvents');
MousedownEvent.initEvent('mousedown', false, true);
// const MouseupEvent = new Event('mouseup', { bubbles: true });
var MouseupEvent = document.createEvent('HTMLEvents');
MouseupEvent.initEvent('mouseup', false, true); //参数名称:事件名称，是否支持冒泡，是否取消默认事件
const fakeClickOutSide = () => {
    document.dispatchEvent(MousedownEvent);
    document.dispatchEvent(MouseupEvent);
    lock = true; // console.log('dispatchEvent');
};
const mousedownHandle = e => {
    let classList = e.target.classList; //永远接收dom事件
    if (classList.contains('el-input__inner') || classList.contains('el-select__caret')) {
        lock = false;
        return;
    }
    if (lock) return;
    fakeClickOutSide();
};
const mousewheelHandle = e => {
    if (lock || e.target.classList.contains('el-select-dropdown__item') || e.target.parentNode.classList.contains('el-select-dropdown__item')) return;
    if (lock || e.target.classList.contains('el-time-spinner__item') || e.target.parentNode.classList.contains('el-time-spinner__item')) return;
    fakeClickOutSide();
};
const eventListener = type => {
    el[type + 'EventListener']('mousedown', mousedownHandle);
    window[type + 'EventListener']('mousewheel', mousewheelHandle);
    window[type + 'EventListener']('DOMMouseScroll', mousewheelHandle); // fireFox 3.5+
};
export default {
    mounted() {
        el = this.$el;
        el.addFakeClickOutSideEventCount = el.addFakeClickOutSideEventCount || 0;
        !el.addFakeClickOutSideEventCount &&
            this.$nextTick(() => {
                eventListener('add');
            });
        el.addFakeClickOutSideEventCount += 1;
    },
    destroyed() {
        eventListener('remove');
        el.addFakeClickOutSideEventCount -= 1;
    }
};

2、将文件导入你想要的页面中并进行混入,操作如下:

import fakeClickOutSide '项目文件地址'
export default {
  mixins: [fakeClickOutSide]
}
